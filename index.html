<script>

const PLATFORMS = ['TikTok','Instagram','Facebook','YouTube'];

const PLAT_COLORS = { TikTok:'#3cf0ff', Instagram:'#ff60c0', Facebook:'#60ffb0', YouTube:'#ffb840' };

const ACC_COLORS  = ['#3cf0ff','#ff60c0','#60ffb0','#ffb840','#b8ff3c','#c060ff'];

let charts = [];


// AUTO-LOAD LOGIC

window.onload = async () => {

    try {

        const response = await fetch('datos.xlsx');

        if (!response.ok) throw new Error();

        const arrayBuffer = await response.arrayBuffer();

        const wb = XLSX.read(arrayBuffer, { type:'array', cellDates:true });

        processWorkbook(wb);

    } catch (e) {

        document.getElementById('status-text').textContent = "Carga Manual";

        document.getElementById('status-sub').innerHTML = "No se halló 'datos.xlsx'. Arrastre el archivo aquí.";

        document.getElementById('fileInput').style.display = 'block';

    }

};


const dropZone = document.getElementById('dropZone');

dropZone.onclick = () => document.getElementById('fileInput').click();

document.getElementById('fileInput').onchange = e => { if(e.target.files[0]) manualProcess(e.target.files[0]); };


function manualProcess(file) {

    const reader = new FileReader();

    reader.onload = ev => {

        const wb = XLSX.read(ev.target.result, { type:'array', cellDates:true });

        processWorkbook(wb);

    };

    reader.readAsArrayBuffer(file);

}


function processWorkbook(wb) {

    const data = parseWorkbook(wb);

    document.getElementById('upload-screen').style.display = 'none';

    document.getElementById('dashboard-screen').style.display = 'block';

    renderDashboard(data);

}


function parseWorkbook(wb) {

  const weeks = [];

  wb.SheetNames.forEach(name => {

    const m = name.match(/semana\s*(\d+)/i);

    if (!m) return;

    const ws = wb.Sheets[name];

    const rows = XLSX.utils.sheet_to_json(ws, { header:1, defval:'' });

    let hRow = -1;

    for (let i = 0; i < rows.length; i++) {

      if (rows[i].some(c => String(c).toLowerCase().includes('tiktok'))) { hRow = i; break; }

    }

    if (hRow < 0) return;

    const h = rows[hRow].map(c => String(c).toLowerCase());

    const ci = { tik: h.findIndex(x => x.includes('tiktok')), ig: h.findIndex(x => x.includes('instagram')), fb: h.findIndex(x => x.includes('facebook')), yt: h.findIndex(x => x.includes('youtube')), tot: h.findIndex(x => x.includes('total')) };

    const accounts = [];

    for (let i = hRow + 2; i < rows.length; i++) {

      const cuenta = String(rows[i][0] || '').trim();

      if (!cuenta || /bono|fecha|total/i.test(cuenta)) continue;

      const tik = pn(rows[i][ci.tik]), ig = pn(rows[i][ci.ig]), fb = pn(rows[i][ci.fb]), yt = pn(rows[i][ci.yt]);

      accounts.push({ name: cuenta, TikTok: tik, Instagram: ig, Facebook: fb, YouTube: yt, total: (tik+ig+fb+yt) });

    }

    weeks.push({ num: parseInt(m[1]), accounts });

  });

  return { weeks: weeks.sort((a,b) => a.num - b.num) };

}


function pn(v) { const n = parseFloat(String(v).replace(/[^0-9.-]/g, '')); return isNaN(n) ? 0 : n; }


function renderDashboard(data) {

  const tabs = document.getElementById('weekTabs');

  tabs.innerHTML = '';

  

  const resumenBtn = document.createElement('button');

  resumenBtn.className = 'tab active';

  resumenBtn.textContent = 'RESUMEN MENSUAL';

  resumenBtn.onclick = () => { selectTab(resumenBtn); renderResumen(data); };

  tabs.appendChild(resumenBtn);


  data.weeks.forEach(w => {

    const btn = document.createElement('button');

    btn.className = 'tab';

    btn.textContent = `SEMANA ${w.num}`;

    btn.onclick = () => { selectTab(btn); renderWeek(w); };

    tabs.appendChild(btn);

  });

  renderResumen(data);

}


function selectTab(btn) { document.querySelectorAll('.tab').forEach(t => t.classList.remove('active')); btn.classList.add('active'); }


function renderResumen(data) {

  destroyCharts();

  const accs = {};

  data.weeks.forEach(w => w.accounts.forEach(a => {

    if(!accs[a.name]) accs[a.name] = { total: 0, history: [] };

    accs[a.name].total += a.total;

    accs[a.name].history.push(a.total);

  }));

  

  const grandTotal = Object.values(accs).reduce((s, x) => s + x.total, 0);

  const names = Object.keys(accs);


  let html = `<div class="metrics-row animate-up">

    <div class="metric-card hero"><div style="font-size:9px; color:var(--muted)">TOTAL MES</div><div class="metric-val">${fmt(grandTotal)}</div></div>

    ${names.slice(0,3).map((n, i) => `<div class="metric-card"><div style="font-size:9px; color:var(--muted)">${n}</div><div class="metric-val" style="font-size:18px">${fmtK(accs[n].total)}</div></div>`).join('')}

  </div>

  <div class="grid-2 animate-up"><div class="panel"><canvas id="mainChart"></canvas></div><div class="panel"><canvas id="pieChart"></canvas></div></div>`;

  

  document.getElementById('dashContent').innerHTML = html;

  

  new Chart(document.getElementById('mainChart'), { type:'line', data:{ labels: data.weeks.map(w=>'S'+w.num), datasets: names.map((n,i)=>({ label:n, data:accs[n].history, borderColor:ACC_COLORS[i%6], tension:0.4 })) }, options: { responsive:true, maintainAspectRatio:false, plugins:{legend:{display:false}} } });

  new Chart(document.getElementById('pieChart'), { type:'doughnut', data:{ labels:names, datasets:[{ data:names.map(n=>accs[n].total), backgroundColor:ACC_COLORS }] }, options: { responsive:true, maintainAspectRatio:false, cutout:'70%' } });

}


function renderWeek(week) {

    destroyCharts();

    const plats = { TikTok:0, Instagram:0, Facebook:0, YouTube:0 };

    week.accounts.forEach(a => PLATFORMS.forEach(p => plats[p] += a[p]));

    const total = Object.values(plats).reduce((a,b)=>a+b,0);


    let html = `<div class="metrics-row animate-up">

        <div class="metric-card hero"><div style="font-size:9px; color:var(--muted)">SEMANA ${week.num}</div><div class="metric-val">${fmt(total)}</div></div>

        ${PLATFORMS.map(p => `<div class="metric-card"><div style="font-size:9px; color:var(--muted)">${p}</div><div class="metric-val" style="font-size:18px">${fmtK(plats[p])}</div></div>`).join('')}

    </div>

    <div class="accounts-grid animate-up">

        ${week.accounts.map(a => `<div class="account-card"><div style="font-weight:700; margin-bottom:10px; border-bottom:1px solid var(--border); padding-bottom:5px;">${a.name}</div><div style="font-size:18px; font-weight:800; color:var(--lime)">${fmt(a.total)}</div></div>`).join('')}

    </div>`;

    document.getElementById('dashContent').innerHTML = html;

}


function fmt(n) { return Math.round(n).toLocaleString('es-AR'); }

function fmtK(n) { return n >= 1000 ? (n/1000).toFixed(1)+'k' : n; }

function destroyCharts() { charts.forEach(c => c.destroy()); charts = []; }
